#include <iostream>
using namespace std;

template <class T>
class Vector {
    T *arr;
    int size;

public:
    Vector(int n) {
        size = n;
        arr = new T[size];
    }

    
    void setValue(int index, T value) {
        if (index >= 0 && index < size)
            arr[index] = value;
        else
            cout << "Index out of range!\n";
    }

   
    void modify(int index, T newValue) {
        if (index >= 0 && index < size)
            arr[index] = newValue;
        else
            cout << "Index out of range!\n";
    }

  
    void multiply(T scalar) {
        for (int i = 0; i < size; i++)
            arr[i] *= scalar;
    }

     
    void display() {
        cout << "(";
        for (int i = 0; i < size; i++) {
            cout << arr[i];
            if (i != size - 1)
                cout << ",";
        }
        cout << ")\n";
    }
};

int main() {
    int n;
    cout << "Enter size of vector: ";
    cin >> n;

    Vector<int> v(n);

    cout << "Enter " << n << " elements:\n";
    for (int i = 0; i < n; i++) {
        int x;
        cin >> x;
        v.setValue(i, x);
    }

    cout << "Original Vector: ";
    v.display();

    int index, value;
    cout << "Enter index to modify and new value: ";
    cin >> index >> value;
    v.modify(index, value);

    cout << "After Modification: ";
    v.display();

    int scalar;
    cout << "Enter scalar to multiply: ";
    cin >> scalar;
    v.multiply(scalar);

    cout << "After Scalar Multiplication: ";
    v.display();

    return 0;
}
