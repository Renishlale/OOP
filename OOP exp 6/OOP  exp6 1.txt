  
#include <iostream>
using namespace std;


class A
{
    public:
    int a;
    public:
    int getA()
    {
        cout<<"Enter the value of a:";
        cin>>a;
        return 0;
    }
};

class B:public A
{
    public:
    int b;
    public:
    int getB()
    {
        cout<<"Enter the value of b:";
        cin>>b;
        return 0;
    }
};

class C:public B
{
 
     
    public:
    int sum()
    {
        
       cout<<"Sum:"<<a+b;
       return 0;
    }
};

int main()
{
   C d;
   d.getA();
   d.getB();
   d.sum();
   
}
