 #include <iostream>
using namespace std;

class Number {
public:
    int num;

    // constructor
    Number(int n = 0) {
        num = n;
    }

    // Overload unary minus operator
    void operator-() {
        num = -num;   // negate the value
    }

    void display() {
        cout << "Value: " << num << endl;
    }
};

int main() {
    int x;
    cout << "Enter any value: ";
    cin >> x;

    Number n(x);

    -n;  // use unary operator

    cout << "After negation: ";
    n.display();

    return 0;
}
