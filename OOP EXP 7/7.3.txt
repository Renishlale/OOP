 #include <iostream>
using namespace std;

class Number {
public:
    int num;
    Number(int n = 0) { num = n; }
    void operator-() { num = -num; }
    void display() { cout << "Value: " << num << endl; }
};

int main() {
    int x;
    cout << "Enter any value: ";
    cin >> x;
    Number n(x);
    -n;
    cout << "After negation: ";
    n.display();
    return 0;
}
